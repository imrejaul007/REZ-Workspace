HTML Drop-in Widget 

<script src="https://cdn.rez.money/schedule/widget.js"></script>
<div id="booking"></div>
<script>
  ReZSchedule.init({
    container: '#booking',
    username: 'drsharma',
    slug: 'consultation'
  });
</script>

SDK JavaScript/TypeScript 

import { createClient } from '@rez/schedule-sdk';

const client = createClient({
  apiKey: 'your-api-key'
});

const { slots } = await client.availability.get({
  username: 'drsharma',
  slug: 'consultation',
  startDate: '2026-05-27',
  endDate: '2026-05-29'
});

await client.bookings.create({
  eventTypeId: slots[0].eventTypeId,
  startTime: slots[0].startTime,
  attendeeName: 'John Doe',
  attendeeEmail: 'john@example.com'
});

Python Django/FastAPI 

from rez_schedule import ReZSchedule

client = ReZSchedule(api_key="your-key")

slots = client.availability.get(
    username="drsharma",
    slug="consultation",
    start_date="2026-05-27",
    end_date="2026-05-29"
)

client.bookings.create(
    event_type_id=slots[0].id,
    start_time=slots[0].start_time,
    attendee_name="John Doe",
    attendee_email="john@example.com"
)
